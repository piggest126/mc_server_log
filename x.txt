# 查看 CPU 支持的 AVX-512 等级
cat /proc/cpuinfo | grep flags | head -1 | grep -oE 'avx512\w+' | sort -u

# 最佳编译（自动检测 CPU）
icx -O3 -xHOST -qopt-zmm-usage=high -flto -o mc_log_filter mc_log_filter.c

# 或指定 AVX-512 具体级别
icx -O3 -march=skylake-avx512 -qopt-zmm-usage=high -flto -o mc_log_filter mc_log_filter.c

# 二进制大小对比
ls -lh mc_log_filter
